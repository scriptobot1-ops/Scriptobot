        min = Math.ceil(min);

      const getRandomFloat = (min, max, decimals) => {
          const str = (Math.random() * (max - min) + min).toFixed(decimals);

      const updateCardNumbers = () => {

        if (!savedNumbers || lastUpdateDate !== today) {
